   197→/*  Suggestion Card                                                    */
   198→/* ------------------------------------------------------------------ */
   199→
   200→function SuggestionCard({ suggestion, index }: { suggestion: Suggestion; index: number }) {
   201→  const { target_post, relevance_score, anchor_text, context_sentence, confidence } =
   202→    suggestion;
   203→
   204→  // Color-code relevance score
   205→  const relevanceColor =
   206→    relevance_score > 0.2
   207→      ? "bg-emerald-500/15 text-emerald-700 dark:text-emerald-400 border-emerald-500/25"
   208→      : relevance_score > 0.1
   209→        ? "bg-amber-500/15 text-amber-700 dark:text-amber-400 border-amber-500/25"
   210→        : "bg-rose-500/15 text-rose-700 dark:text-rose-400 border-rose-500/25";
   211→
   212→  // Color-code confidence
   213→  const confidenceColor =
   214→    confidence === "high"
   215→      ? "bg-teal-500/15 text-teal-700 dark:text-teal-400 border-teal-500/25"
   216→      : confidence === "medium"
   217→        ? "bg-amber-500/15 text-amber-700 dark:text-amber-400 border-amber-500/25"
   218→        : "bg-rose-500/15 text-rose-700 dark:text-rose-400 border-rose-500/25";
   219→
   220→  // Highlight anchor text in context sentence
   221→  const renderContext = () => {
   222→    const parts = context_sentence.split(new RegExp(`(${escapeRegex(anchor_text)})`, "gi"));
   223→    return parts.map((part, i) =>
   224→      part.toLowerCase() === anchor_text.toLowerCase() ? (
   225→        <strong key={i} className="font-bold text-amber-900 dark:text-amber-200">
   226→          {part}
   227→        </strong>
   228→      ) : (
   229→        part
   230→      )
   231→    );
   232→  };
   233→
   234→  return (
   235→    <motion.div
   236→      custom={index}
   237→      variants={cardVariants}
   238→      initial="hidden"
   239→      animate="visible"
   240→    >
   241→      <Card className="border-border/60 hover:border-border hover:shadow-md transition-shadow">
   242→        <CardContent className="p-4 sm:p-5 space-y-3">
   243→          {/* Target Post */}
   244→          <div className="flex items-start justify-between gap-3">
   245→            <div className="min-w-0">
   246→              <a
   247→                href={`#${target_post.slug}`}
   248→                className="text-sm font-semibold text-foreground hover:text-teal-600 dark:hover:text-teal-400 transition-colors flex items-center gap-1.5 group"
   249→              >
   250→                <ExternalLink className="size-3.5 shrink-0 opacity-50 group-hover:opacity-100 transition-opacity" />
   251→                <span className="truncate">{target_post.title}</span>
   252→              </a>
   253→              <p className="text-xs text-muted-foreground mt-1 line-clamp-2">
   254→                {target_post.excerpt}
   255→              </p>
   256→            </div>
   257→          </div>
   258→
   259→          {/* Badges Row */}
   260→          <div className="flex flex-wrap items-center gap-2">
   261→            <Badge variant="outline" className={cn("text-xs", relevanceColor)}>
   262→              <BarChart3 className="size-3 mr-1" />
   263→              {relevance_score.toFixed(2)} relevance
   264→            </Badge>
   265→            <Badge variant="outline" className={cn("text-xs capitalize", confidenceColor)}>
   266→              <Target className="size-3 mr-1" />
   267→              {confidence}
   268→            </Badge>
   269→          </div>
   270→
   271→          {/* Anchor Text Highlight */}
   272→          <div className="rounded-lg bg-amber-50 dark:bg-amber-500/10 border border-amber-200/60 dark:border-amber-500/20 p-3 sm:p-4">
   273→            <p className="text-[10px] uppercase tracking-wider font-semibold text-amber-700 dark:text-amber-400 mb-1.5 flex items-center gap-1.5">
   274→              <Zap className="size-3" />
   275→              Suggested Anchor
   276→            </p>
   277→            <p className="text-sm sm:text-base leading-relaxed text-foreground/90">
   278→              &ldquo;{renderContext()}&rdquo;
   279→            </p>
   280→          </div>
   281→        </CardContent>
   282→      </Card>
   283→    </motion.div>
   284→  );
   285→}

   299→export function DemoSection() {
   300→  const [title, setTitle] = useState("");
   301→  const [content, setContent] = useState("");
   302→  const [selectedSample, setSelectedSample] = useState<string>("");
   303→  const [pipeline, setPipeline] = useState<PipelineStep>("idle");
   304→  const [results, setResults] = useState<SuggestResponse | null>(null);
   305→  const [error, setError] = useState<string | null>(null);
   306→  const [engineLive, setEngineLive] = useState(false);
   307→
   308→  const wordCount = content.trim() ? content.trim().split(/\s+/).length : 0;
   309→  const charCount = content.length;
   310→
   311→  /* ---- Check engine status ---- */
   312→  useEffect(() => {
   313→    const checkEngine = async () => {
   314→      try {
   315→        const res = await fetch("/?XTransformPort=3030/posts");
   316→        if (res.ok) setEngineLive(true);
   317→      } catch {
   318→        setEngineLive(false);
   319→      }
   320→    };
   321→    checkEngine();
   322→    const interval = setInterval(checkEngine, 15000);
   323→    return () => clearInterval(interval);
   324→  }, []);
   325→
   326→  /* ---- Load sample post ---- */
   327→  const handleSampleSelect = useCallback((value: string) => {
   328→    setSelectedSample(value);
   329→    const sample = SAMPLE_POSTS.find((s) => s.label === value);
   330→    if (sample) {
   331→      setTitle(sample.title);
   332→      setContent(sample.content);
   333→      // Reset results when loading new content
   334→      setResults(null);
   335→      setError(null);
   336→      setPipeline("idle");
   337→    }
   338→  }, []);
   339→
   340→  /* ---- Generate suggestions ---- */
   341→  const handleGenerate = useCallback(async () => {
   342→    if (!content.trim()) return;
   343→
   344→    // Reset
   345→    setError(null);
   346→    setResults(null);
   347→    setPipeline("chunking");
   348→
   349→    // Step 1: Chunking (simulate 1.2s)
   350→    await new Promise((r) => setTimeout(r, 1200));
   351→    setPipeline("retrieval");
   352→
   353→    // Step 2: Retrieval (simulate 1s)
   354→    await new Promise((r) => setTimeout(r, 1000));
   355→    setPipeline("generation");
   356→
   357→    // Step 3: Generation — fire API in parallel with simulated delay
   358→    try {
   359→      const res = await fetch("/?XTransformPort=3030/suggest", {
   360→        method: "POST",
   361→        headers: { "Content-Type": "application/json" },
   362→        body: JSON.stringify({ text: content, title: title || undefined }),
   363→      });
   364→
   365→      if (!res.ok) {
   366→        throw new Error(`Server returned ${res.status}: ${res.statusText}`);
   367→      }
   368→
   369→      const data: SuggestResponse = await res.json();
   370→      // Small delay to let generation animation play
   371→      await new Promise((r) => setTimeout(r, 600));
   372→      setPipeline("done");
   373→      setResults(data);
   374→    } catch (err) {
   375→      setPipeline("idle");
   376→      setError(
   377→        err instanceof Error
   378→          ? err.message
   379→          : "An unexpected error occurred. Please check that the demo engine is running and try again."
   380→      );
   381→    }
   382→  }, [content, title]);
   383→
   384→  /* ---- Try Again ---- */
   385→  const handleRetry = useCallback(() => {
   386→    setError(null);
   387→    setPipeline("idle");
   388→  }, []);
   389→
   390→  /* ---- Pipeline step status helper ---- */
   391→  const stepStatus = (step: "chunking" | "retrieval" | "generation") => {
   392→    if (pipeline === "idle") return "pending" as const;
   393→    const order: PipelineStep[] = ["chunking", "retrieval", "generation", "done"];
   394→    const currentIdx = order.indexOf(pipeline);
   395→    const stepIdx = order.indexOf(step);
   396→    if (stepIdx < currentIdx) return "done" as const;
   397→    if (stepIdx === currentIdx) return "active" as const;
   398→    return "pending" as const;
   399→  };
   400→
   401→  return (
   402→    <section id="demo" className="scroll-mt-24 py-20 sm:py-28">
   403→      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
   404→        {/* ---- Header ---- */}
   405→        <motion.div
   406→          initial={{ opacity: 0, y: 20 }}
   407→          whileInView={{ opacity: 1, y: 0 }}
   408→          viewport={{ once: true, margin: "-60px" }}
   409→          transition={{ duration: 0.5 }}
   410→          className="text-center mb-12"
   411→        >
   412→          <div className="flex items-center justify-center gap-3 mb-4">
   413→            <h2 className="text-3xl sm:text-4xl lg:text-5xl font-bold tracking-tight text-foreground">
   414→              Try It Now — Live Demo
   415→            </h2>
   416→            {/* ENGINE LIVE badge */}
   417→            <span className="relative flex h-3 w-3">
   418→              {engineLive && (
   419→                <span className="animate-ping absolute inline-flex h-full w-full rounded-full bg-teal-400 opacity-75" />
   420→              )}
   421→              <span
   422→                className={cn(
   423→                  "relative inline-flex rounded-full h-3 w-3",
   424→                  engineLive ? "bg-teal-500" : "bg-muted-foreground/40"
   425→                )}
   426→              />
   427→            </span>
   428→            <Badge
   429→              variant="outline"
   430→              className={cn(
   431→                "text-xs font-semibold px-2.5 py-0.5 transition-colors",
   432→                engineLive
   433→                  ? "border-teal-500/40 bg-teal-500/10 text-teal-700 dark:text-teal-400"
   434→                  : "border-muted-foreground/25 bg-muted text-muted-foreground"
   435→              )}
   436→            >
   437→              {engineLive ? "ENGINE LIVE" : "OFFLINE"}
   438→            </Badge>
   439→          </div>
   440→          <p className="mt-3 text-muted-foreground text-base sm:text-lg max-w-2xl mx-auto">
   441→            Paste any blog post and watch LinkForge generate real internal link
   442→            suggestions in real-time.
   443→          </p>
   444→        </motion.div>
   445→
   446→        {/* ---- Two-Column Layout ---- */}
   447→        <motion.div
   448→          variants={containerVariants}
   449→          initial="hidden"
   450→          whileInView="visible"
   451→          viewport={{ once: true, margin: "-40px" }}
   452→          className="grid grid-cols-1 lg:grid-cols-2 gap-6 lg:gap-8 items-start"
   453→        >
   454→          {/* ========================= */}
   455→          {/* Left Column — Input        */}
   456→          {/* ========================= */}
   457→          <div className="space-y-5">
   458→            {/* Sample Post Selector */}
   459→            <div className="space-y-2">
   460→              <label className="text-sm font-medium text-foreground flex items-center gap-1.5">
   461→                <FileText className="size-4 text-muted-foreground" />
   462→                Or try a sample post:

   464→              <Select value={selectedSample} onValueChange={handleSampleSelect}>
   465→                <SelectTrigger className="w-full">
   466→                  <SelectValue placeholder="Select a sample blog post..." />

   468→                <SelectContent>
   469→                  {SAMPLE_POSTS.map((s) => (
   470→                    <SelectItem key={s.label} value={s.label}>
   471→                      {s.label}
   472→                    </SelectItem>
   473→                  ))}
   474→                </SelectContent>
   475→              </Select>
   476→            </div>
   477→
   478→            {/* Title Input */}
   479→            <div className="space-y-2">
   480→              <label
   481→                htmlFor="demo-title"
   482→                className="text-sm font-medium text-foreground"
   483→              >

   486→              </label>
   487→              <Input
   488→                id="demo-title"
   489→                placeholder="e.g., My Latest Blog Post"

   496→            {/* Content Textarea */}

   535→                pipeline === "idle" &&
   536→                  content.trim() &&

   539→                  "bg-muted text-muted-foreground shadow-none cursor-not-allowed"
   540→              )}
   541→            >
   542→              {pipeline !== "idle" ? (
   543→                <>
   544→                  <Loader2 className="size-5 animate-spin" />
   545→                  Analyzing...
   546→                </>
   547→              ) : (
   548→                <>
   549→                  <Sparkles className="size-5" />
   550→                  Generate Link Suggestions

   555→          </div>
   556→
   557→          {/* ========================= */}

   559→          {/* ========================= */}
   560→          <div className="min-h-[400px]">
   561→            <AnimatePresence mode="wait">
   562→              {/* ---- Pipeline Progress ---- */}
   563→              {(pipeline === "chunking" ||
   564→                pipeline === "retrieval" ||
   565→                pipeline === "generation") && (
   566→                <motion.div
   567→                  key="pipeline"

   575→                      <h3 className="text-base font-semibold text-foreground flex items-center gap-2">
   576→                        <Wand2 className="size-4 text-teal-600 dark:text-teal-400" />
   577→                        Processing Pipeline
   578→                      </h3>
   579→                    </CardHeader>
   580→                    <CardContent className="space-y-5">
   581→                      <PipelineStepCard
   582→                        step={1}
   583→                        icon={Scissors}
   584→                        label="Chunking"
   585→                        status={stepStatus("chunking")}
   586→                        result={

   589→                            : "Splitting into paragraphs..."
   590→                        }
   591→                      />
   592→                      <div className="border-l-2 border-dashed border-muted-foreground/20 ml-4 h-3" />

   595→                        icon={Search}

   602→                        }
   603→                      />
   604→                      <div className="border-l-2 border-dashed border-muted-foreground/20 ml-4 h-3" />

   606→                        step={3}
   607→                        icon={PenLine}

   609→                        status={stepStatus("generation")}
   610→                        result={

   612→                            ? `${results.suggestions.length} suggestions ready`
   613→                            : "Generating anchor text..."
   614→                        }
   615→                      />
   616→                    </CardContent>
   617→                  </Card>
   618→                </motion.div>
   619→              )}
   620→
   621→              {/* ---- Error State ---- */}
   622→              {error && pipeline === "idle" && (
   623→                <motion.div
   624→                  key="error"
   625→                  initial={{ opacity: 0, scale: 0.96 }}
   626→                  animate={{ opacity: 1, scale: 1 }}
   627→                  exit={{ opacity: 0 }}
   628→                  transition={{ duration: 0.3 }}
   629→                >
   630→                  <Card className="border-rose-500/40 bg-rose-500/5">
   631→                    <CardContent className="p-6 flex flex-col items-center text-center gap-4">
   632→                      <div className="flex items-center justify-center w-14 h-14 rounded-full bg-rose-500/10">
   633→                        <AlertCircle className="size-7 text-rose-500" />
   634→                      </div>
   635→                      <div>
   636→                        <h3 className="text-base font-semibold text-foreground mb-1">
   637→                          Something went wrong
   638→                        </h3>
   639→                        <p className="text-sm text-muted-foreground max-w-sm">
   640→                          {error}
   641→                        </p>
   642→                      </div>
   643→                      <Button
   644→                        variant="outline"
   645→                        onClick={handleRetry}
   646→                        className="border-rose-500/30 text-rose-700 dark:text-rose-400 hover:bg-rose-500/10 gap-2"
   647→                      >
   648→                        <ArrowRight className="size-4 rotate-[-135deg]" />

   651→                    </CardContent>
   652→                  </Card>
   653→                </motion.div>
   654→              )}
   655→
   656→              {/* ---- Results Display ---- */}
   657→              {results && pipeline === "done" && (
   658→                <motion.div
   659→                  key="results"
   660→                  initial={{ opacity: 0 }}
   661→                  animate={{ opacity: 1 }}
   662→                  exit={{ opacity: 0 }}
   663→                  transition={{ duration: 0.4 }}
   664→                  className="space-y-4"
   665→                >
   666→                  <div className="flex items-center justify-between">
   667→                    <h3 className="text-base font-semibold text-foreground flex items-center gap-2">
   668→                      <Link2 className="size-4 text-teal-600 dark:text-teal-400" />
   669→                      Link Suggestions
   670→                    </h3>
   671→                    <Badge
   672→                      variant="outline"
   673→                      className="border-teal-500/30 bg-teal-500/10 text-teal-700 dark:text-teal-400 text-xs"

   679→                  {/* Suggestion Cards */}
   680→                  <div className="space-y-4 max-h-[600px] overflow-y-auto pr-1">

   682→                      {results.suggestions.map((s, i) => (
   683→                        <SuggestionCard key={s.target_post.id} suggestion={s} index={i} />
   684→                      ))}
   685→                    </AnimatePresence>
   686→                  </div>
   687→
   688→                  {/* Processing Stats */}
   689→                  <Card className="bg-muted/40 border-border/40">
   690→                    <CardContent className="p-3 sm:p-4">
   691→                      <div className="flex flex-wrap items-center justify-center gap-3 sm:gap-5 text-xs text-muted-foreground">
   692→                        <span className="flex items-center gap-1.5">
   693→                          <Clock className="size-3.5" />
   694→                          <span className="font-medium text-foreground">
   695→                            {results.processing_time_ms}ms
   696→                          </span>{" "}
   697→                          processing time
   698→                        </span>
   699→                        <span className="hidden sm:inline text-muted-foreground/40">|</span>
   700→                        <span className="flex items-center gap-1.5">
   701→                          <BarChart3 className="size-3.5" />
   702→                          <span className="font-medium text-foreground">
   703→                            {results.chunks_count}
   704→                          </span>{" "}
   705→                          chunks analyzed
   706→                        </span>
   707→                        <span className="hidden sm:inline text-muted-foreground/40">|</span>
   708→                        <span className="flex items-center gap-1.5">
   709→                          <Target className="size-3.5" />
   710→                          <span className="font-medium text-foreground">
   711→                            {results.suggestions.length}
   712→                          </span>{" "}
   713→                          suggestions
   714→                        </span>
   715→                      </div>
   716→                    </CardContent>
   717→                  </Card>
   718→                </motion.div>
   719→              )}
   720→
   721→              {/* ---- Empty State ---- */}
   722→              {!error && pipeline === "idle" && !results && (
   723→                <motion.div
   724→                  key="empty"
   725→                  initial={{ opacity: 0 }}
   726→                  animate={{ opacity: 1 }}
   727→                  exit={{ opacity: 0 }}
   728→                  transition={{ duration: 0.3 }}
   729→                >
   730→                  <Card className="border-dashed border-border/60 bg-muted/20 h-full min-h-[400px] flex items-center justify-center">
   731→                    <CardContent className="p-8 flex flex-col items-center text-center gap-4">
   732→                      <div className="flex items-center justify-center w-20 h-20 rounded-2xl bg-muted/60">
   733→                        <Link2 className="size-10 text-muted-foreground/30" />
   734→                      </div>
   735→                      <div>
   736→                        <p className="text-base font-semibold text-foreground/70">
   737→                          Your link suggestions will appear here
   738→                        </p>
   739→                        <p className="text-sm text-muted-foreground mt-1.5 max-w-xs mx-auto">
   740→                          Paste content above or select a sample post to get started
   741→                        </p>
   742→                      </div>
   743→                    </CardContent>
   744→                  </Card>
   745→                </motion.div>
   746→              )}
   747→            </AnimatePresence>
   748→          </div>
   749→        </motion.div>
   750→      </div>
   751→    </section>
   752→  );

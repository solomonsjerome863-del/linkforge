    26→/* ─── Fade-in on scroll wrapper ─── */
    27→function FadeIn({
    28→  children,
    29→  className,
    30→  delay = 0,
    31→}: {
    32→  children: React.ReactNode;
    33→  className?: string;
    34→  delay?: number;
    35→}) {
    36→  return (
    37→    <motion.div
    38→      initial={{ opacity: 0, y: 30 }}
    39→      whileInView={{ opacity: 1, y: 0 }}
    40→      viewport={{ once: true, margin: "-60px" }}

    42→      className={className}
    43→    >
    44→      {children}
    45→    </motion.div>
    46→  );
    47→}

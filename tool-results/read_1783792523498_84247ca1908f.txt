    53→/* ─── Scroll animation wrapper ─── */
    54→function FadeIn({
    55→  children,
    56→  className,
    57→  delay = 0,
    58→}: {
    59→  children: React.ReactNode;
    60→  className?: string;
    61→  delay?: number;
    62→}) {
    63→  return (
    64→    <motion.div
    65→      initial={{ opacity: 0, y: 30 }}
    66→      whileInView={{ opacity: 1, y: 0 }}
    67→      viewport={{ once: true, margin: "-60px" }}
    68→      transition={{ duration: 0.6, delay, ease: "easeOut" }}
    69→      className={className}
    70→    >
    71→      {children}
    72→    </motion.div>
    73→  );
    74→}

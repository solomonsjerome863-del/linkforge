   320→        let finalChunk = currentChunk.trim();
   321→        if (lastHeading) {
   322→          const context = `## ${lastHeading}`;
   323→          if (context.length <= 500) {
   324→            finalChunk = context + '\n\n' + finalChunk;
   325→          }
   326→        }
   327→        chunks.push(finalChunk);
   328→        currentChunk = '';
   329→      }
   330→    } else {
   331→      currentChunk += (currentChunk ? '\n' : '') + line;
   332→    }
   333→  }
   334→
   335→  // Don't forget the last chunk
   336→  if (currentChunk.trim()) {
   337→    let finalChunk = currentChunk.trim();
   338→    if (lastHeading) {
   339→      const context = `## ${lastHeading}`;
   340→      if (context.length <= 500) {
   341→        finalChunk = context + '\n\n' + finalChunk;
   342→      }
   343→    }
   344→    chunks.push(finalChunk);
   345→  }
   346→
   347→  return chunks.filter((c) => c.trim().length > 0);
   348→}

   350→// ============================================================
   351→// RELEVANCE SCORING: TF overlap (simulates vector search)
   352→// ============================================================
   353→function tokenize(text: string): string[] {
   354→  return text
   355→    .toLowerCase()
   356→    .replace(/[^a-z0-9\s]/g, ' ')
   357→    .split(/\s+/)
   358→    .filter((w) => w.length >= 3 && !STOP_WORDS.has(w));
   359→}
   360→
   361→function buildFreqMap(tokens: string[]): Map<string, number> {
   362→  const freq = new Map<string, number>();
   363→  for (const t of tokens) {
   364→    freq.set(t, (freq.get(t) || 0) + 1);
   365→  }
   366→  return freq;
   367→}
   368→
   369→export function computeRelevance(textA: string, textB: string): number {
   370→  const tokensA = tokenize(textA);
   371→  const tokensB = tokenize(textB);
   372→
   373→  if (tokensA.length === 0 || tokensB.length === 0) return 0;
   374→
   375→  const freqA = buildFreqMap(tokensA);
   376→  const freqB = buildFreqMap(tokensB);
   377→
   378→  let overlap = 0;
   379→  for (const [word, countB] of freqB) {
   380→    const countA = freqA.get(word);
   381→    if (countA !== undefined) {
   382→      overlap += Math.min(countA, countB);
   383→    }
   384→  }
   385→
   386→  const maxTotal = Math.max(tokensA.length, tokensB.length);
   387→  return overlap / maxTotal;
   388→}

   414→export function generateFallbackAnchor(title: string): string {
   415→  let anchor = title
   416→    .replace(/^(the|a|an|complete|ultimate|comprehensive|how to|your)\s+/i, '')
   417→    .replace(/[:\-–—]/g, '')
   418→    .trim();
   419→  const words = anchor.split(/\s+/);
   420→  if (words.length > 6) {
   421→    anchor = words.slice(0, 6).join(' ');
   422→  }
   423→  return anchor.toLowerCase();
   424→}

   426→export function findBestMatchingChunk(
   427→  chunks: string[],
   428→  targetTitle: string,
   429→  targetExcerpt: string
   430→): string {
   431→  let bestChunk = chunks[0];
   432→  let bestScore = -1;
   433→  const targetText = `${targetTitle} ${targetExcerpt}`;
   434→
   435→  for (const chunk of chunks) {
   436→    const score = computeRelevance(chunk, targetText);
   437→    if (score > bestScore) {
   438→      bestScore = score;
   439→      bestChunk = chunk;
   440→    }
   441→  }
   442→  return bestChunk;
   443→}

   445→export async function generateAnchorText(
   446→  sourceChunk: string,
   447→  targetTitle: string,
   448→  targetExcerpt: string
   449→): Promise<{ sentence: string; anchor_text: string } | null> {
   450→  // Smart fallback: derive natural anchor from target post title
   451→  // In production, this is replaced by z-ai-web-dev-sdk LLM call for AI-crafted anchors
   452→  const fallback = generateFallbackAnchor(targetTitle);
   453→  const sentences = sourceChunk
   454→    .replace(/#{1,3}\s+.+/g, '')
   455→    .split(/[.!?]+/)
   456→    .map((s) => s.trim())
   457→    .filter((s) => s.length > 20);
   458→
   459→  // Find the sentence most related to the target
   460→  const targetWords = new Set(
   461→    targetTitle.toLowerCase().split(/\s+/).filter((w) => w.length > 3 && !STOP_WORDS.has(w))
   462→  );
   463→  
   464→  let bestSentence = sentences[0] || sourceChunk.slice(0, 200);
   465→  let bestOverlap = 0;
   466→  for (const s of sentences) {
   467→    const sWords = new Set(s.toLowerCase().split(/\s+/).filter((w) => w.length > 3 && !STOP_WORDS.has(w)));
   468→    let overlap = 0;
   469→    for (const w of targetWords) { if (sWords.has(w)) overlap++; }
   470→    if (overlap > bestOverlap) {
   471→      bestOverlap = overlap;
   472→      bestSentence = s;
   473→    }
   474→  }
   475→
   476→  return {
   477→    sentence: bestSentence,
   478→    anchor_text: fallback
   479→  };
   480→}
